\Section{Permutations}{}

\begin{theorem}
    Any permutation \( \sigma \in S_n \) can be written as a product of
    transpositions.
\end{theorem}

\begin{theorem}[HW]
    Let \( \sigma \in S_n \), and let \( \inv(\sigma) \) be the number of
    inversions of the permutation \( \sigma \). Then, \( \sgn(\sigma) =
    (-1)^{\inv(\sigma)} \), and \( \sgn(\tau \sigma) = -\sgn(\sigma) \), where
    \( \tau \in S_n \) is a transposition.
\end{theorem}

\begin{theorem}[HW]
    Let \( \sigma \in S_n \) be a \( k \)-cycle. Then, \( \sgn(\sigma) = (-1)^{k+1} \).
\end{theorem}
