\Section{Adjoints}{}

\begin{definition}
    In an inner product space \( V \), for \( T : V \to W \) a linear transformation, a linera transformation \( S : W \to V \) is an
    adjoint for \( T \) if for all \( \vec{v} \in V, \vec{w} \in W \),
    \[
        \inner{\vec{w}}{T(\vec{v})}_W = \inner{S(\vec{w})}{\vec{v}}_V
    .\]
\end{definition}

\begin{theorem}
    The adjoint is unique.
\end{theorem}

\begin{theorem}
    \( \Ker(T^* T) = \Ker(T) \).
\end{theorem}

\begin{theorem}
    \( (T^*)^* = T \).
\end{theorem}

\begin{theorem}
    The adjoint always exists for finite dimensional inner product spaces.
\end{theorem}
