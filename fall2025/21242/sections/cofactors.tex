\Section{Cofactors}{}

\begin{theorem}
    Let \( A \in \F^{n \times n} \) and let \( A_{\overline{ij}} \) denote \( A \) with its \( i \)th row and \( j \)th column removed. Then,
    \begin{align*}
        \det(A) &= \sum_{j = 1}^n (-1)^{j + 1} b_{ij} \det(B_{\overline{ij}}) \\
        &= \sum_{i = 1}^n (-1)^{i + 1} b_{ij} \det(B_{\overline{ij}}) \\
    .\end{align*}
\end{theorem}

\begin{definition}
    Given \( B \in \F^{n \times n} \), define \( C_B \) as the cofactor matrix via
    \[
        (C_B)_{ij} = (-1)^{i+j} \det(B_{\overline{ij}})
    .\]
\end{definition}

\begin{theorem}
    If \( B \in \F^{n \times n} \) is invertible, then
    \[
        B^{-1} = \frac{1}{\det(B)} C_B^T
    .\]
    Equivalently,
    \[
        B C^T_B = \det(B) I_n
    .\]
\end{theorem}
